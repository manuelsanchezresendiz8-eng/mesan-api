import re
from dataclasses import dataclass
from typing import List, Literal, Optional

Severity = Literal["CRITICA", "ALTA", "MEDIA"]

@dataclass
class Contradiction:
    descripcion: str
    severidad: Severity
    impacto: str
    accion: str

class ContradictionGate:
    REQUIRED_FIELDS = ["CONTRADICCION", "SEVERIDAD", "IMPACTO", "ACCION"]
    SEVERITY_MAP = {"CRITICA": 3, "ALTA": 2, "MEDIA": 1}

    def parse(self, text: str) -> List[Contradiction]:
        blocks = self._split_blocks(text)
        return [c for b in blocks for c in [self._parse_block(b)] if c]

    def _split_blocks(self, text: str) -> List[str]:
        return [b.strip() for b in text.split("\n\n") if b.strip()]

    def _parse_block(self, block: str) -> Optional[Contradiction]:
        if "SIN_CONTRADICCIONES" in block.upper(): return None
        fields = {}
        for line in block.split("\n"):
            if ":" not in line: continue
            k, v = line.split(":", 1)
            fields[k.strip().upper()] = v.strip()
        if not all(f in fields for f in self.REQUIRED_FIELDS): return None
        return Contradiction(descripcion=fields["CONTRADICCION"],
                             severidad=fields["SEVERIDAD"].upper(),
                             impacto=fields["IMPACTO"], accion=fields["ACCION"])

    def validate(self, contradictions: List[Contradiction]) -> dict:
        if not contradictions: return {"status": "PASS", "reason": "SIN_CONTRADICCIONES"}
        critical = [c for c in contradictions if c.severidad == "CRITICA"]
        if critical: return {"status": "BLOCK", "reason": "CONTRADICCIONES_CRITICAS_DETECTADAS", "count": len(critical)}
        return {"status": "FAIL", "reason": "CONTRADICCIONES_DETECTADAS", "count": len(contradictions)}

    def run(self, text: str) -> dict:
        contradictions = self.parse(text)
        return {"result": self.validate(contradictions), "contradictions": contradictions}
